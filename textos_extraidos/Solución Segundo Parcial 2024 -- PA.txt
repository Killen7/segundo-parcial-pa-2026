class Subject { 
    public: 
        void notifyObservers(Event * ev); 
        void addObserver(Observer * obs); 
        void removeObserver(Observer * obs); 
        void clearObservers(); 
        virtual ~Subject() {}; 
        virtual void onCancel(Observer * obs) = 0; 
    private: 
        ICollection * observers; 
}; 
void Subject::notifyObservers(Event * ev){ 
    IIterator * it = this->observers->getIterator(); 
    Observer * o; 
    while (it->hasCurrent()) { 
        o = (Observer *) it->current(); 
        o->notify(ev); 
        if(ev->isCancelled()){ 
            this->onCancel(o); 
            break; 
        } 
        it->next(); 
    } 
    delete it; 
} 
void Subject::addObserver(Observer * obs){ 
    this->observers->add(obs); 
} 
void Subject::removeObserver(Observer * obs){ 
    this->observers->remove(obs); 
} 
void Subject::clearObservers(){ 
    IIterator * it = this->observers->getIterator(); 
    while (it->hasCurrent()) { 
        this->remove(it->current()); 
        it->next(); 
    } 
    delete it; 
} 


class Observer: public ICollectible { 
    public: 
        virtual void notify(Event * ev) = 0; 
        virtual ~Observer() {}; 
};

class Tienda: public Observer { 
    public: 
        string getNombre(); 
        void notify(Event * ev); 
    private: 
        IDictionary * productos; 
        string nombre; 
}; 


void Tienda::notify(Event * ev) { 
    EventoCompra * ec = (EventoCompra *) ev; 
    IKey * key = new KeyInt (ec->getIdProd()); 
    Producto * p = (Producto *) this->productos->find(key); 
    if(p != NULL && (p->getStock() >= ec->getCant())) { 
        int nuevoStock = p->getStock() - ec->getCant(); 
        p->setStock(nuevoStock); 
        ev->cancel(); 
    } 
    delete key; 
}
